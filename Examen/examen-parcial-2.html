// 1. Calcular promedio de un estudiante
function calcularPromedio(estudiante) {
    const suma = estudiante.notas.reduce((acc, nota) => acc + nota, 0);
    return suma / estudiante.notas.length;
}

// 2. Obtener estudiantes aprobados (promedio >= 14)
function estudiantesAprobados() {
    return estudiantes
        .filter(est => calcularPromedio(est) >= 14)
        .map(est => est.nombre);
}

// 3. Encontrar el mejor estudiante
function mejorEstudiante() {
    return estudiantes.reduce((mejor, actual) => {
        return calcularPromedio(actual) > calcularPromedio(mejor) ? actual : mejor;
    });
}

// 4. Calcular promedio general
function promedioGeneral() {
    const sumaPromedios = estudiantes
        .map(est => calcularPromedio(est))
        .reduce((acc, prom) => acc + prom, 0);
    return sumaPromedios / estudiantes.length;
}

# Forma tradicional cuadrados = [] for x in range(5): cuadrados.append(x ** 2) # [0, 1, 4, 9, 16] # Con list comprehension cuadrados = [x ** 2 for x in range(5)] # [0, 1, 4, 9, 16] # Con condici√≥n (filtrado) pares = [x for x in range(10) if x % 2 == 0] # [0, 2, 4, 6, 8] # Transformar elementos nombres = ["ana", "juan", "mar√≠a"] mayusculas = [n.upper() for n in nombres] # ["ANA", "JUAN", "MAR√çA"] # Con if-else (ternario) clasificacion = ["par" if x % 2 == 0 else "impar" for x in range(5)] # ["par", "impar", "par", "impar", "par"] # Anidado (matriz) matriz = [[j for j in range(3)] for i in range(3)] # [[0,1,2], [0,1,2], [0,1,2]] # Aplanar matriz aplanado = [elem for fila in matriz for elem in fila] # [0, 1, 2, 0, 1, 2, 0, 1, 2]

üìã Cuantificadores

Repeticiones del elemento anterior

import re # Cuantificadores r"*" # 0 o m√°s veces r"+" # 1 o m√°s veces r"?" # 0 o 1 vez (opcional) r"{n}" # Exactamente n veces r"{n,}" # Al menos n veces r"{n,m}" # Entre n y m veces # Ejemplos re.search(r"go*d", "gd") # Coincide (0 'o') re.search(r"go*d", "good") # Coincide (2 'o') re.search(r"go+d", "gd") # NO coincide (necesita al menos 1) re.search(r"go+d", "good") # Coincide re.search(r"colou?r", "color") # Coincide (u opcional) re.search(r"colou?r", "colour") # Coincide # Cuantificadores exactos re.fullmatch(r"\d{4}", "2025") # Coincide re.fullmatch(r"\d{2,4}", "25") # Coincide re.fullmatch(r"\d{2,4}", "2025") # Coincide re.findall(r"\d{2,}", "1 25 2025") # ['25', '2025'] # Greedy vs Non-greedy texto = "contenido" re.search(r"<.*>", texto).group() # "contenido" (greedy) re.search(r"<.*?>", texto).group() # "" (non-greedy)

import threading import time # Funci√≥n que ejecutar√° cada hilo def descargar(url): print(f"Descargando {url}...") time.sleep(2) # Simula descarga print(f"Completado: {url}") return f"OK: {url}" # Crear hilos urls = ["url1.com", "url2.com", "url3.com"] hilos = [] inicio = time.time() for url in urls: hilo = threading.Thread(target=descargar, args=(url,)) hilos.append(hilo) hilo.start() # Inicia el hilo # Esperar a que todos terminen for hilo in hilos: hilo.join() print(f"Total: {time.time() - inicio:.2f}s") # ~2 segundos en lugar de ~6 # Con ThreadPoolExecutor (m√°s moderno) from concurrent.futures import ThreadPoolExecutor with ThreadPoolExecutor(max_workers=3) as executor: resultados = list(executor.map(descargar, urls)) # El GIL (Global Interpreter Lock) limita a un hilo # ejecutando Python a la vez, pero libera durante I/O
